SELECT INSTRUCTORES.IdInstructor, INSTRUCTORES.Nombre & ' ' & INSTRUCTORES.Apellido_Paterno & ' ' & INSTRUCTORES.Apellido_Materno AS INSTRUCTOR, CUPONES.FolioCupon, CONCEPTO_INGRESOS.IdConcepto, CONCEPTO_INGRESOS.Descripcion, CUPONES.ImporteAPagar, CUPONES.FechaAlta, CUPONES.FechaAplicacion, CUPONES.FechaPago, USUARIOS_CLUB.NoFamilia, USUARIOS_CLUB.Nombre & ' ' &  USUARIOS_CLUB.A_Paterno  & ' ' & USUARIOS_CLUB.A_Materno AS USUARIO FROM ((CUPONES LEFT JOIN CONCEPTO_INGRESOS ON CUPONES.IdConcepto = CONCEPTO_INGRESOS.IdConcepto) LEFT JOIN INSTRUCTORES ON CUPONES.IdInstructorAplicacion = INSTRUCTORES.IdInstructor) LEFT JOIN USUARIOS_CLUB ON CUPONES.IdMember = USUARIOS_CLUB.IdMember WHERE (((CUPONES.FechaAplicacion) Is Not Null) AND ((CUPONES.FechaPago)=<@Fecha>)) ORDER BY INSTRUCTORES.IdInstructor, CONCEPTO_INGRESOS.IdConcepto
